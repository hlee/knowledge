### good random vs bad random 

```ruby
class BadRandom
  def initialize(seed)
    @last = seed & 0xFFFFFFFF
  end
 
  def rand(max = nil)
    shifted = @last << 1
    highbit = @last >> 31
    @last = (shifted | highbit) & 0xFFFFFFFF
    max ? (@last % max) : @last
  end
end
 
br = BadRandom.new(0xFFFF)
"%032b" % br.rand               # => "00000000000000011111111111111110"
```
